// 5. FONCTION DE CALCUL DES DISTANCES
// ===========================================

function createDistanceRaster(features, maxDistance) {
  // Ajouter une propri√©t√© constante pour la conversion en raster
  var featuresWithValue = features.map(function(f) {
    return f.set('constant', 1);
  });
  
  // Convertir les features en image
  var image = featuresWithValue.reduceToImage({
    properties: ['constant'],  
    reducer: ee.Reducer.first()
  }).unmask();
  
  // Calculer la distance euclidienne en m√®tres
  var distance = image.fastDistanceTransform({
    neighborhood: 256,
    units: 'pixels',
    metric: 'squared_euclidean'
  }).sqrt().multiply(ee.Image.pixelArea().sqrt());
  
  return distance.rename('distance');
}

// Fonction pour calculer les stats (mean, min, max)
function calculateStats(distanceRaster, protectedAreas, infrastructureName) {
  var stats = distanceRaster.reduceRegions({
    collection: protectedAreas,
    reducer: ee.Reducer.mean()
      .combine({reducer2: ee.Reducer.min(), sharedInputs: true})
      .combine({reducer2: ee.Reducer.max(), sharedInputs: true}),
    scale: 100,
    crs: 'EPSG:4326'
  });
  
  return stats;
}

// Fonction CORRIG√âE pour afficher les stats globales
function displayGlobalStats(stats, infrastructureName) {
  // R√©cup√©rer les valeurs avec evaluate() pour √©viter l'affichage des objets ee.Number
  stats.aggregate_mean('mean').evaluate(function(meanVal) {
    stats.aggregate_min('min').evaluate(function(minVal) {
      stats.aggregate_max('max').evaluate(function(maxVal) {
        print("--- " + infrastructureName + " ---");
        if (meanVal !== null) {
          print("  Distance moyenne : " + (meanVal/1000).toFixed(2) + " km");
          print("  Distance minimale : " + (minVal/1000).toFixed(2) + " km");
          print("  Distance maximale : " + (maxVal/1000).toFixed(2) + " km");
        } else {
          print("  Donn√©es non disponibles");
        }
      });
    });
  });
}
